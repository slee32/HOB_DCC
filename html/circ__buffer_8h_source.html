236 bool circ_buffer<qType, qSize>::jam (qType data)
237 {
238  // Write the data and move the write pointer to the next element
239  buffer[i_put] = data;
240  if (++i_put >= qSize)
241  {
242  i_put = 0;
243  }
244 
245  // Check if the buffer is already full; if so, the read index has to be moved so
246  // that it points to the oldest unread data, which isn't the data written now
